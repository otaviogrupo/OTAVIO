import { createClient } from "@supabase/supabase-js"; const supabase = createClient( process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_ROLE ); export default async function handler(req, res) { // =============================== // 🔐 VERIFICAÇÃO META // =============================== if (req.method === "GET") { const VERIFY_TOKEN = process.env.VERIFY_TOKEN; const mode = req.query["hub.mode"]; const token = req.query["hub.verify_token"]; const challenge = req.query["hub.challenge"]; if (mode === "subscribe" && token === VERIFY_TOKEN) { return res.status(200).send(challenge); } return res.status(403).send("Erro de verificação"); } // =============================== // 📩 RECEBER MENSAGEM // =============================== if (req.method === "POST") { try { const body = req.body; const change = body.entry?.[0]?.changes?.[0]?.value; // ignora eventos vazios if (!change) return res.status(200).send("OK"); // ignora status (delivered, read etc) if (change.statuses) return res.status(200).send("OK"); const msg = change.messages?.[0]; if (!msg) return res.status(200).send("OK"); const telefone = msg.from; const mensagem = msg.text?.body || "[mensagem não textual]"; const phoneId = change.metadata?.phone_number_id; console.log("📩 Mensagem de:", telefone); console.log("📱 Phone ID:", phoneId); // =============================== // 💾 SALVAR MENSAGEM (CLIENTE) // =============================== await supabase.from("conversas_whatsapp").insert({ telefone, mensagem, role: "user", message_id: msg.id, status: "received" }); // =============================== // 🤖 CHAMAR OPENAI // =============================== let resposta = "Tive um problema ao responder, pode tentar novamente? 😊"; try { const ai = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${process.env.OPENAI_API_KEY}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "gpt-4o-mini", messages: [ { role: "system", content: "Você é um atendente do Mercatto Delícia. Responda de forma simpática, rápida e objetiva." }, { role: "user", content: mensagem } ] }) }).then(r => r.json()); resposta = ai?.choices?.[0]?.message?.content || resposta; } catch (err) { console.log("❌ ERRO OPENAI:", err); } // =============================== // 💬 ENVIAR RESPOSTA WHATSAPP // =============================== await fetch(`https://graph.facebook.com/v19.0/${phoneId}/messages`, { method: "POST", headers: { "Authorization": `Bearer ${process.env.WHATSAPP_TOKEN}`, "Content-Type": "application/json" }, body: JSON.stringify({ messaging_product: "whatsapp", to: telefone, type: "text", text: { body: resposta } }) }); // =============================== // 💾 SALVAR RESPOSTA (BOT) // =============================== await supabase.from("conversas_whatsapp").insert({ telefone, mensagem: resposta, role: "assistant", status: "sent" }); return res.status(200).send("OK"); } catch (err) { console.error("❌ ERRO GERAL:", err); return res.status(500).json({ erro: err.message }); } } }
